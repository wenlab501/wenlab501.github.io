par(family="JH",mar=c(1,1,1,1)) # 
#st_area
town['mazhou_density']=town$mazhou_count/st_area(town)*10^6
plot(st_geometry(county),bg = "lightblue1") #
choroLayer(town,var='mazhou_density',method="quantile",nclass=5,col=carto.pal(pal1 = "red.pal", n1 = 5),legend.values.rnd=3,add=T,legend.title.txt ='')
layoutLayer(title="",north=T,postitle = "center") 

#method
#fisher-jenksquantileequal
methods_name=c("","","")
methods=c("fisher-jenks","quantile","equal")
par(mfrow=c(1,3),mar=c(2,0,2,0))  
for(i in 1:3){
  plot(st_geometry(county),bg = "lightblue1") 
  choroLayer(town,var='mazhou_density',method=methods[i],nclass=5,col=carto.pal(pal1 = "red.pal", n1 = 5),legend.values.rnd=3,add=T,legend.title.txt ='')
  layoutLayer(title=methods_name[i],north=T,postitle = "center") 
}
